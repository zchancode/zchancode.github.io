0x03_Android实现压缩图片 Android工具类
2023-04-20 - 2023/04/20/0x03_android/

这个工具类可以实现压缩图片

import static android.os.Environment.DIRECTORY_DOWNLOADS;

import android.graphics.Bitmap;
import android.graphics.BitmapFactory;
import android.os.Environment;

import java.io.ByteArrayOutputStream;
import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;

public class PhotoCompresser {
    /**
     *
     * @param originalFile 原始图片文件
     * @param compressRatio 压缩率，1-100之间的整数
     * @return 压缩后的图片文件
     */
    public static File compress(File originalFile, int compressRatio) {
        File sdFile = Environment.getExternalStoragePublicDirectory(DIRECTORY_DOWNLOADS);
        BitmapFactory.Options options = new BitmapFactory.Options();
        options.inJustDecodeBounds = true;
        Bitmap emptyBitmap = BitmapFactory.decodeFile(originalFile.getAbsolutePath(), options);
        options.inJustDecodeBounds = false;
        Bitmap resultBitmap = BitmapFactory.decodeFile(originalFile.getAbsolutePath(), options);
        ByteArrayOutputStream bos = new ByteArrayOutputStream();
        resultBitmap.compress(Bitmap.CompressFormat.JPEG, compressRatio, bos);
        File resultFile = new File(sdFile, "resultImg.jpg");
        try {
            FileOutputStream fos = new FileOutputStream(resultFile);
            fos.write(bos.toByteArray());
            fos.flush();
            fos.close();
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        } catch (IOException e) {
            e.printStackTrace();
        }
        return resultFile;
    }
}

以下是如何使用compress方法的示例代码：

File originalFile = new File(Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DCIM), "IMG_20220825_163257.jpg");
File resultFile = PhotoCompresser.compress(originalFile, 80);

// 在控制台上打印结果路径
if(resultFile.exists()) {
    System.out.println("New image file: " + resultFile.getAbsolutePath());
} else {
    System.out.println("Image file compression failed...");
}

上面的代码将通过指定压缩率(80)，压缩原始文件(IMG_20220825_163257.jpg)，并返回一个压缩后的文件。
